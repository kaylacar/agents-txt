Capability: search-recent
  Endpoint: https://api.x.com/2/tweets/search/recent
  Method: GET
  Protocol: REST
  Auth: bearer-token
  Auth-Endpoint: https://api.x.com/2/oauth2/token
  Auth-Docs: https://developer.x.com/en/docs/authentication/oauth-2-0
  Rate-Limit: 180/minute
  Scopes: tweet.read,users.read
  Description: Search posts from the last 7 days by keyword, hashtag, user mention, or operator query
  Param: query (query, string, required) - Search query using X query language operators
  Param: max_results (query, integer) - Results per page, 10-100, default 10
  Param: next_token (query, string) - Pagination cursor from previous response
  Param: start_time (query, string) - Earliest post timestamp (ISO 8601)
  Param: end_time (query, string) - Latest post timestamp (ISO 8601)
  OpenAPI: https://api.x.com/2/openapi.json

# --- Per-agent policies -------------------------------------------------------
#
# Tier mapping:
#   Default (*) -> read-only, conservative limits (equivalent to Free tier)
#   Named agents -> negotiated access matching their approved use case
#
# To be listed as a named agent, register at developer.x.com and apply for
# Elevated or Pro access. Bot accounts must comply with the Automation Rules:
# https://help.x.com/en/using-x/automated-account-rules

# Default: any unidentified agent gets read-only search
Agent: *
  Rate-Limit: 60/minute
  Capabilities: search-recent, get-tweet, get-user-by-username

# Customer support bots - reply to mentions, no mass-follow, no DMs without prior contact
Agent: acme-support-bot
  Rate-Limit: 300/minute
  Capabilities: search-recent, get-tweet, get-user-by-username, get-mentions, get-user-timeline, post-tweet, reply-to-tweet, like-tweet
  Agent-Declaration: https://bot.acme.com/.well-known/agents.txt

# News and media aggregators - high read throughput, post their own content
Agent: newsbot-wire
  Rate-Limit: 500/minute
  Capabilities: search-recent, get-tweet, get-user-by-username, get-user-timeline, get-mentions, post-tweet, filtered-stream, sampled-stream

# Brand monitoring / social listening - high read, no write
Agent: brandwatch-monitor
  Rate-Limit: 1000/minute
  Capabilities: search-recent, get-tweet, get-user-by-username, get-user-timeline, get-mentions, filtered-stream, sampled-stream

# Research agents - academic access, read-only
Agent: research-agent
  Rate-Limit: 500/minute
  Capabilities: search-recent, get-tweet, get-user-by-username, get-user-timeline, get-mentions, get-likes, sampled-stream
